      if (data.rotation?.length)     setRotation(data.rotation);

      if (data.setupDone)            setSetupDone(true);

      if (data.parentPin)            setParentPin(data.parentPin);

 

      // Last-used member stored in localStorage per uid (device-local, not synced)

      try { const w = localStorage.getItem(`dh-who-${uid}`); if (w) setWho(w); } catch {}

 

      // Real-time listeners for frequently-changing data

      checksOff = fRef('checks').on('value',  s => setChecksData(s.val() || {}));

      pointsOff = fRef('points').on('value',  s => setPointsData(s.val() || {}));

 

      setLoaded(true);

    }).catch(e => { setError('Failed to load: '+e.message); setLoaded(true); });

 

    return () => {

      if (checksOff) fRef('checks').off('value', checksOff);

      if (pointsOff) fRef('points').off('value', pointsOff);

    };

  }, [uid]);

 

  // ── Debounced meta write — 800ms after last change

  const debouncedMeta = useMemo(() => debounce((m, kh, ph, r, pin) => {

    fRef('').update({ members:m, kidHabits:kh, parentHabits:ph, rotation:r,

                      parentPin:pin, schemaVersion:SCHEMA_VERSION, updatedAt:Date.now() });

  }, 800), [uid]);

  useEffect(() => () => debouncedMeta.cancel(), [debouncedMeta]);

 

  const saveMeta = useCallback((m, kh, ph, r, pin=parentPin) => {

    setMembers(m); setKidHabits(kh); setParentHabits(ph); setRotation(r);

    if (pin !== undefined) setParentPin(pin);

    debouncedMeta(m, kh, ph, r, pin);

  }, [debouncedMeta, parentPin]);

 

  const saveWho = useCallback(w => {

    setWho(w);

    try { if (w) localStorage.setItem(`dh-who-${uid}`, w);

          else   localStorage.removeItem(`dh-who-${uid}`); } catch {}

  }, [uid]);

 

  const finishSetup = useCallback((m, kh, ph, r, pin) => {

    setSetupDone(true);

    saveMeta(m, kh, ph, r, pin);

    fRef('setupDone').set(true);

  }, [saveMeta]);

 

  // ── Optimistic habit toggle with Firebase rollback

  const toggleHabit = useCallback((name, tid) => {

    const day    = dayKey();

    const sk     = toKey(name);

    const cur    = checksData[name]?.[day]?.[tid] || false;

    const next   = !cur;

 

    // Immediate local update (optimistic)

    setChecksData(prev => ({

      ...prev,

      [name]: { ...(prev[name]||{}), [day]: { ...(prev[name]?.[day]||{}), [tid]:next } }

    }));

 

    // Points logic for children

    const m = members.find(x => x.name===name);

    if (m?.role==='child') {

      const delta = next ? POINTS_PER : -POINTS_PER;

      fRef(`points/${sk}`).transaction(cur => Math.max(0,(cur||0)+delta));

 

      if (next) {

        const flat    = getFlatHabits(name, members, kidHabits, parentHabits, rotation);

        const allDone = flat.every(t => t.id===tid ? true : (checksData[name]?.[day]?.[t.id]||false));

        if (allDone) {

          fRef(`points/${sk}`).transaction(cur => (cur||0)+POINTS_BONUS);

          const ck = name+day;

          if (!celebRef.current[ck]) { celebRef.current[ck]=true; setTimeout(burst,200); }

        }

      }

    }

 

    // Firebase write — rollback UI if it fails

    fRef(`checks/${sk}/${day}/${tid}`).set(next).catch(() => {

      setChecksData(prev => ({

        ...prev,

        [name]: { ...(prev[name]||{}), [day]: { ...(prev[name]?.[day]||{}), [tid]:cur } }

      }));

    });

  }, [checksData, members, kidHabits, parentHabits, rotation, uid]);

 

  // ── Derived helpers

  const isChecked = useCallback((name,tid) =>

    checksData[name]?.[dayKey()]?.[tid] || false

  , [checksData]);

 

  const doneCount = useCallback(name =>

    getFlatHabits(name, members, kidHabits, parentHabits, rotation)

      .filter(t => isChecked(name,t.id)).length

  , [members, kidHabits, parentHabits, rotation, isChecked]);

 

  const getPoints = useCallback(name =>

    pointsData[toKey(name)] || 0

  , [pointsData]);

 

  const getStreak = useCallback(name => {

    const today = new Date();

    const cd    = checksData[name] || {};

    const tot   = getFlatHabits(name, members, kidHabits, parentHabits, rotation).length;

    if (tot===0) return 0;

    let streak = 0;

    for (let i=1; i<365; i++) {

      const d = new Date(today); d.setDate(d.getDate()-i);

      const k = d.toISOString().slice(0,10);

      const dayChecks = cd[k];

      if (!dayChecks) break;

      if (Object.values(dayChecks).filter(v=>v===true).length >= Math.floor(tot*.8)) streak++;

      else break;

    }

    const todayDone = getFlatHabits(name, members, kidHabits, parentHabits, rotation)

      .filter(t=>isChecked(name,t.id)).length;

    if (todayDone===tot && tot>0) streak++;

    return streak;

  }, [checksData, members, kidHabits, parentHabits, rotation, isChecked]);

 

  const getHistory = useCallback(name => {

    const cd  = checksData[name] || {};

    const tot = getFlatHabits(name, members, kidHabits, parentHabits, rotation).length;

    return Object.keys(cd)

      .filter(k => /^\d{4}-\d{2}-\d{2}$/.test(k))

      .sort().reverse().slice(0,30)

      .map(k => {

        const done = Object.values(cd[k]).filter(v=>v===true).length;

        return { date:k, done, total:tot, pct:calcPct(done,tot) };

      });

  }, [checksData, members, kidHabits, parentHabits, rotation]);

 

  // ── 7 PM notification — with cleanup to prevent memory leak

  useEffect(() => {

    if (!who) return;

    const flat = getFlatHabits(who, members, kidHabits, parentHabits, rotation);

    if (flat.length===0) return;

    const target = new Date(); target.setHours(19,0,0,0);

    if (target <= new Date()) target.setDate(target.getDate()+1);

    const tid = setTimeout(async () => {

      const remaining = flat.length - doneCount(who);

      if (remaining > 0 && Notification.permission==='granted') {

        new Notification('Daily Habits',{

          body:`${remaining} habit${remaining>1?'s':''} still to go, ${who}!`,

          icon:'/icon-192.png'

        });

      }

    }, target-Date.now());

    return () => clearTimeout(tid); // ← prevents duplicate notifications

  }, [who, members, kidHabits, parentHabits, rotation]);

 

  const value = {

    checksData, pointsData, members, kidHabits, parentHabits, rotation, parentPin,

    who, setupDone, loaded, error, online,

    setMembers, setKidHabits, setParentHabits, setRotation,

    saveWho, finishSetup, saveMeta, toggleHabit,

    isChecked, doneCount, getPoints, getStreak, getHistory,

    signOut: () => { saveWho(null); auth.signOut(); },

    userEmail: user.email,

  };

 

  return <DataCtx.Provider value={value}>{children}</DataCtx.Provider>;

}

  );

class ErrorBoundary extends React.Component {

  constructor(p) { super(p); this.state={error:null}; }

  static getDerivedStateFromError(e) { return {error:e}; }

  render() {

    if (this.state.error) return (

      <div style={{...S.shell,display:'flex',alignItems:'center',justifyContent:'center',minHeight:'100vh'}}>

        <div style={{textAlign:'center',padding:'0 24px',maxWidth:360}}>

          <h1 style={{...S.serif(24),marginBottom:12}}>something went wrong</h1>

          <p style={{fontSize:13,color:'#bfb5a2',lineHeight:1.6,marginBottom:20}}>{this.state.error?.message}</p>

          <button onClick={()=>this.setState({error:null})} style={S.btnP}>try again</button>

        </div>

      </div>

    );

    return this.props.children;

  }

}

function DataProvider({ user, children }) {

  const uid  = user.uid;

  const [pointsData,   setPointsData]   = useState({});

  const [setupDone,    setSetupDone]    = useState(false);

  const [parentPin,    setParentPin]    = useState('');

  const [loaded,       setLoaded]       = useState(false);

  const [error,        setError]        = useState(null);

  const [online,       setOnline]       = useState(true);

  const celebRef = useRef({});

 

  // ── Online / offline

  useEffect(() => {

    const on  = () => setOnline(true);

    const off = () => setOnline(false);

    window.addEventListener('online',  on);

    window.addEventListener('offline', off);

    return () => { window.removeEventListener('online',on); window.removeEventListener('offline',off); };

  }, []);

 

  // ── Initial load + migration + real-time listeners

  useEffect(() => {

    let checksOff = null;

    let pointsOff = null;

 

    fRef('').once('value').then(snap => {

      const raw  = snap.val() || {};

      const data = migrateData(raw);

 

      // Persist migration if version changed

      if (data.schemaVersion !== raw.schemaVersion) {

        fRef('').update({ schemaVersion: data.schemaVersion });

      }

 

      if (data.members?.length)      setMembers(data.members);

      if (data.kidHabits?.length)    setKidHabits(data.kidHabits);

      if (data.parentHabits?.length) setParentHabits(data.parentHabits);

      if (data.rotation?.length)     setRotation(data.rotation);

      if (data.setupDone)            setSetupDone(true);

      if (data.parentPin)            setParentPin(data.parentPin);

 

      // Last-used member stored in localStorage per uid (device-local, not synced)

      try { const w = localStorage.getItem(`dh-who-${uid}`); if (w) setWho(w); } catch {}

 

      // Real-time listeners for frequently-changing data

      checksOff = fRef('checks').on('value',  s => setChecksData(s.val() || {}));

      pointsOff = fRef('points').on('value',  s => setPointsData(s.val() || {}));

 

      setLoaded(true);

    }).catch(e => { setError('Failed to load: '+e.message); setLoaded(true); });

 

    return () => {

      if (checksOff) fRef('checks').off('value', checksOff);

      if (pointsOff) fRef('points').off('value', pointsOff);

    };

  }, [uid]);

 

  // ── Debounced meta write — 800ms after last change

  const debouncedMeta = useMemo(() => debounce((m, kh, ph, r, pin) => {

    fRef('').update({ members:m, kidHabits:kh, parentHabits:ph, rotation:r,

                      parentPin:pin, schemaVersion:SCHEMA_VERSION, updatedAt:Date.now() });

  }, 800), [uid]);

  useEffect(() => () => debouncedMeta.cancel(), [debouncedMeta]);

 

  const saveMeta = useCallback((m, kh, ph, r, pin=parentPin) => {

    setMembers(m); setKidHabits(kh); setParentHabits(ph); setRotation(r);

    if (pin !== undefined) setParentPin(pin);

    debouncedMeta(m, kh, ph, r, pin);

  }, [debouncedMeta, parentPin]);

 

  const saveWho = useCallback(w => {

    setWho(w);

    try { if (w) localStorage.setItem(`dh-who-${uid}`, w);

          else   localStorage.removeItem(`dh-who-${uid}`); } catch {}

  }, [uid]);

 

  const finishSetup = useCallback((m, kh, ph, r, pin) => {

    setSetupDone(true);

    saveMeta(m, kh, ph, r, pin);

    fRef('setupDone').set(true);

  }, [saveMeta]);

 

  // ── Optimistic habit toggle with Firebase rollback

  const toggleHabit = useCallback((name, tid) => {

    const day    = dayKey();

    const sk     = toKey(name);

    const cur    = checksData[name]?.[day]?.[tid] || false;

    const next   = !cur;

 

    // Immediate local update (optimistic)

    setChecksData(prev => ({

      ...prev,

      [name]: { ...(prev[name]||{}), [day]: { ...(prev[name]?.[day]||{}), [tid]:next } }

    }));

 

    // Points logic for children

    const m = members.find(x => x.name===name);

    if (m?.role==='child') {

      const delta = next ? POINTS_PER : -POINTS_PER;

      fRef(`points/${sk}`).transaction(cur => Math.max(0,(cur||0)+delta));

 

      if (next) {

        const flat    = getFlatHabits(name, members, kidHabits, parentHabits, rotation);

        const allDone = flat.every(t => t.id===tid ? true : (checksData[name]?.[day]?.[t.id]||false));

        if (allDone) {

          fRef(`points/${sk}`).transaction(cur => (cur||0)+POINTS_BONUS);

          const ck = name+day;

          if (!celebRef.current[ck]) { celebRef.current[ck]=true; setTimeout(burst,200); }

        }

      }

    }

 

    // Firebase write — rollback UI if it fails

    fRef(`checks/${sk}/${day}/${tid}`).set(next).catch(() => {

      setChecksData(prev => ({

        ...prev,

        [name]: { ...(prev[name]||{}), [day]: { ...(prev[name]?.[day]||{}), [tid]:cur } }

      }));

    });

  }, [checksData, members, kidHabits, parentHabits, rotation, uid]);

 

  // ── Derived helpers

  const isChecked = useCallback((name,tid) =>

    checksData[name]?.[dayKey()]?.[tid] || false

  , [checksData]);

 

  const doneCount = useCallback(name =>

    getFlatHabits(name, members, kidHabits, parentHabits, rotation)

      .filter(t => isChecked(name,t.id)).length

  , [members, kidHabits, parentHabits, rotation, isChecked]);

 

  const getPoints = useCallback(name =>

    pointsData[toKey(name)] || 0

  , [pointsData]);

 

  const getStreak = useCallback(name => {

    const today = new Date();

    const cd    = checksData[name] || {};

    const tot   = getFlatHabits(name, members, kidHabits, parentHabits, rotation).length;

    if (tot===0) return 0;

    let streak = 0;

    for (let i=1; i<365; i++) {

      const d = new Date(today); d.setDate(d.getDate()-i);

      const k = d.toISOString().slice(0,10);

      const dayChecks = cd[k];

      if (!dayChecks) break;

      if (Object.values(dayChecks).filter(v=>v===true).length >= Math.floor(tot*.8)) streak++;

      else break;

    }

    const todayDone = getFlatHabits(name, members, kidHabits, parentHabits, rotation)

      .filter(t=>isChecked(name,t.id)).length;

    if (todayDone===tot && tot>0) streak++;

    return streak;

  }, [checksData, members, kidHabits, parentHabits, rotation, isChecked]);

 

  const getHistory = useCallback(name => {

    const cd  = checksData[name] || {};

    const tot = getFlatHabits(name, members, kidHabits, parentHabits, rotation).length;

    return Object.keys(cd)

      .filter(k => /^\d{4}-\d{2}-\d{2}$/.test(k))

      .sort().reverse().slice(0,30)

      .map(k => {

        const done = Object.values(cd[k]).filter(v=>v===true).length;

        return { date:k, done, total:tot, pct:calcPct(done,tot) };

      });

  }, [checksData, members, kidHabits, parentHabits, rotation]);

 

  // ── 7 PM notification — with cleanup to prevent memory leak

  useEffect(() => {

    if (!who) return;

    const flat = getFlatHabits(who, members, kidHabits, parentHabits, rotation);

    if (flat.length===0) return;

    const target = new Date(); target.setHours(19,0,0,0);

    if (target <= new Date()) target.setDate(target.getDate()+1);

    const tid = setTimeout(async () => {

      const remaining = flat.length - doneCount(who);

      if (remaining > 0 && Notification.permission==='granted') {

        new Notification('Daily Habits',{

          body:`${remaining} habit${remaining>1?'s':''} still to go, ${who}!`,

          icon:'/icon-192.png'

        });

      }

    }, target-Date.now());

    return () => clearTimeout(tid); // ← prevents duplicate notifications

  }, [who, members, kidHabits, parentHabits, rotation]);

 

  const value = {

    checksData, pointsData, members, kidHabits, parentHabits, rotation, parentPin,

    who, setupDone, loaded, error, online,

    setMembers, setKidHabits, setParentHabits, setRotation,

    saveWho, finishSetup, saveMeta, toggleHabit,

    isChecked, doneCount, getPoints, getStreak, getHistory,

    signOut: () => { saveWho(null); auth.signOut(); },

    userEmail: user.email,

  };

 

  return <DataCtx.Provider value={value}>{children}</DataCtx.Provider>;

}

class ErrorBoundary extends React.Component {

  constructor(p) { super(p); this.state={error:null}; }

  static getDerivedStateFromError(e) { return {error:e}; }

  render() {

    if (this.state.error) return (

      <div style={{...S.shell,display:'flex',alignItems:'center',justifyContent:'center',minHeight:'100vh'}}>

        <div style={{textAlign:'center',padding:'0 24px',maxWidth:360}}>

          <h1 style={{...S.serif(24),marginBottom:12}}>something went wrong</h1>

          <p style={{fontSize:13,color:'#bfb5a2',lineHeight:1.6,marginBottom:20}}>{this.state.error?.message}</p>

          <button onClick={()=>this.setState({error:null})} style={S.btnP}>try again</button>

        </div>

      </div>

    );

    return this.props.children;

  }

}
